#version 400

in vec3 textureCoords;  // notice these are 3D
out vec4 out_Color;


uniform samplerCube cubeMap;  // notice this is a sampleCube
uniform vec3 fogColour;

const float upperLimit = 30.0; // slightly above the horizon
const float lowerLimit = 0.0; // center of the skybox

void main(void){
    vec4 finalColour = texture(cubeMap, textureCoords);

    float factor  = (textureCoords.y - lowerLimit) / (upperLimit - lowerLimit);
    factor = clamp(factor, 0.0, 1.0);
    out_Color = mix(vec4(fogColour, 1.0), finalColour, factor);
}
